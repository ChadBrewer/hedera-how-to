Hedera Client Configuration

Now we'll create the core client that will connect to Hedera's networks. This is the foundation of all DApp interactions.

Step 8: Create Client Configuration

// src/client.js
const { Client, AccountId, PrivateKey } = require('@hashgraph/sdk');
require('dotenv').config();
class HederaClient {
constructor() {
this.client = null;
this.operatorId = null;
this.operatorKey = null;
}
initialize() {
try {
// Get credentials from environment
this.operatorId = AccountId.fromString(process.env.OPERATOR_ID);
this.operatorKey = PrivateKey.fromString(process.env.OPERATOR_KEY);
// Create client based on network
const network = process.env.NETWORK || 'testnet';
if (network === 'mainnet') {
this.client = Client.forMainnet();
} else if (network === 'previewnet') {
this.client = Client.forPreviewnet();
} else {
this.client = Client.forTestnet();
}
// Set operator
this.client.setOperator(this.operatorId, this.operatorKey);
console.log(`‚úÖ Hedera client initialized for ${network}`);
console.log(`üë§ Operator: ${this.operatorId.toString()}`);
} catch (error) {
console.error('‚ùå Failed to initialize Hedera client:', error.message);
throw error;
}
}
getClient() {
return this.client;
}
close() {
if (this.client) {
this.client.close();
}
}
}
module.exports = HederaClient;
Key Features
  • ‚Ä¢ Automatic network detection (testnet/mainnet/previewnet)
  • ‚Ä¢ Secure credential handling via environment variables
  • ‚Ä¢ Error handling and logging
  • ‚Ä¢ Resource cleanup with close() method
Core DApp Functionality

Now we'll create the main DApp logic. This will be a simple "Hello World" application that demonstrates key Hedera features.

Step 9: Create Transaction Handler

// src/transactions.js
const {
AccountBalanceQuery,
TransferTransaction,
Hbar,
TopicCreateTransaction,
TopicMessageSubmitTransaction
} = require('@hashgraph/sdk');
class TransactionHandler {
constructor(client) {
this.client = client;
}
// Check account balance
async checkBalance(accountId) {
try {
const balance = await new AccountBalanceQuery()
.setAccountId(accountId)
.execute(this.client);
console.log(`üí∞ Balance for ${accountId}: ${balance.hbars.toString()}`);
return balance;
} catch (error) {
console.error('‚ùå Error checking balance:', error.message);
throw error;
}
}
// Transfer HBAR
async transferHbar(fromAccount, toAccount, amount) {
try {
const tx = await new TransferTransaction()
.addHbarTransfer(fromAccount, Hbar.from(-amount))
.addHbarTransfer(toAccount, Hbar.from(amount))
.execute(this.client);
const receipt = await tx.getReceipt(this.client);
console.log(`‚úÖ Transfer successful: ${tx.transactionId.toString()}`);
return receipt;
} catch (error) {
console.error('‚ùå Transfer failed:', error.message);
throw error;
}
}
// Create HCS topic
async createTopic(topicMemo) {
try {
const tx = await new TopicCreateTransaction()
.setTopicMemo(topicMemo || 'My first DApp topic')
.execute(this.client);
const receipt = await tx.getReceipt(this.client);
console.log(`‚úÖ Topic created: ${receipt.topicId.toString()}`);
return receipt.topicId;
} catch (error) {
console.error('‚ùå Topic creation failed:', error.message);
throw error;
}
}
// Send message to topic
async sendMessage(topicId, message) {
try {
const tx = await new TopicMessageSubmitTransaction()
.setTopicId(topicId)
.setMessage(message)
.execute(this.client);
const receipt = await tx.getReceipt(this.client);
console.log(`‚úÖ Message sent: ${tx.transactionId.toString()}`);
return receipt;
} catch (error) {
console.error('‚ùå Message sending failed:', error.message);
throw error;
}
}
}
module.exports = TransactionHandler;
Transaction Types Covered
  • ‚Ä¢ AccountBalanceQuery: Check account balances
  • ‚Ä¢ TransferTransaction: Send HBAR between accounts
  • ‚Ä¢ TopicCreateTransaction: Create HCS topics
  • ‚Ä¢ TopicMessageSubmitTransaction: Send messages to topics
Main Application Entry Point

Let's create the main application that ties everything together. This will demonstrate a complete DApp workflow.

Step 10: Create Main Application

// src/index.js
const HederaClient = require('./client');
const TransactionHandler = require('./transactions');
const { AccountId } = require('@hashgraph/sdk');
const chalk = require('chalk');
class MyFirstDApp {
constructor() {
this.hederaClient = new HederaClient();
this.transactionHandler = null;
this.topicId = null;
}
async initialize() {
console.log(chalk.cyan('üöÄ Starting My First DApp...\n'));
// Initialize Hedera client
await this.hederaClient.initialize();
// Create transaction handler
this.transactionHandler = new TransactionHandler(this.hederaClient.getClient());
}
async runDemo() {
try {
// Step 1: Check initial balance
console.log(chalk.yellow('üìä Step 1: Checking account balance...'));
await this.transactionHandler.checkBalance(this.hederaClient.operatorId);
// Step 2: Create HCS topic
console.log(chalk.yellow('\nüí¨ Step 2: Creating HCS topic...'));
this.topicId = await this.transactionHandler.createTopic('Hello World DApp Topic');
// Step 3: Send messages to topic
console.log(chalk.yellow('\nüì® Step 3: Sending messages to topic...'));
const messages = [
'Hello from my first DApp!',
'This message is stored on Hedera',
'Decentralized and secure!'
];
for (let i = 0; i < messages.length; i++) {
console.log(` Sending message ${i + 1}...`);
await this.transactionHandler.sendMessage(this.topicId, messages[i]);
console.log(` ‚úÖ Message ${i + 1} sent!`);
}
// Step 4: Check final balance
console.log(chalk.yellow('\nüí∞ Step 4: Checking final balance...'));
await this.transactionHandler.checkBalance(this.hederaClient.operatorId);
// Success message
console.log(chalk.green('\nüéâ Congratulations! Your first DApp is working!'));
console.log(chalk.blue(`üîó View your topic on HashScan: https://hashscan.io/testnet/topic/${this.topicId.toString()}`));
} catch (error) {
console.error(chalk.red('\nüí• DApp execution failed:'), error.message);
}
}
async shutdown() {
console.log(chalk.cyan('\nüëã Shutting down DApp...'));
await this.hederaClient.close();
}
}
// Main execution
async function main() {
const dapp = new MyFirstDApp();
try {
// Initialize DApp
await dapp.initialize();
// Run demo
await dapp.runDemo();
} catch (error) {
console.error('Fatal error:', error);
} finally {
// Always cleanup
await dapp.shutdown();
}
}
// Run if called directly
if (require.main === module) {
main();
}
module.exports = MyFirstDApp;
