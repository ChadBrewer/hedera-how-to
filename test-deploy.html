DApp Testing Strategies

Testing DApps requires a multi-layered approach. From unit tests to integration testing and mainnet deployment, we'll cover all the essential strategies.

Unit Testing

Test individual functions and components in isolation

  • • Jest or Mocha frameworks
  • • Mock Hedera transactions
  • • Test error handling
  • • Coverage reporting

Integration Testing

Test complete workflows on testnet

  • • Real network interactions
  • • End-to-end workflows
  • • Transaction verification
  • • HashScan validation

Security Testing

Identify vulnerabilities before deployment

  • • Key management audit
  • • Access control testing
  • • Transaction validation
  • • Error handling review
Testing Best Practices
  • • Never use mainnet credentials in tests
  • • Always test with small amounts first
  • • Verify transactions on HashScan before proceeding
  • • Test error conditions and edge cases
  • • Document expected vs actual behavior

const HederaClient = require('../../src/client');
const { Client } = require('@hashgraph/sdk');
describe('HederaClient', () => {
let client;
beforeEach(() => {
client = new HederaClient();
});
describe('initialization', () => {
test('should initialize with testnet by default', async () => {
// Mock environment variables
process.env.OPERATOR_ID = '0.0.123456';
process.env.OPERATOR_KEY = '302e020100300506032b657004220420...';
await client.initialize();
expect(client.hederaClient).toBeDefined();
expect(client.operatorId).toBeDefined();
});
test('should handle missing environment variables', async () => {
// Clear environment variables
delete process.env.OPERATOR_ID;
delete process.env.OPERATOR_KEY;
await expect(client.initialize()).rejects.toThrow();
});
});
describe('network selection', () => {
test('should use mainnet when specified', async () => {
process.env.NETWORK = 'mainnet';
process.env.OPERATOR_ID = '0.0.123456';
process.env.OPERATOR_KEY = '302e020100300506032b657004220420...';
await client.initialize();
// Verify client is configured for mainnet
expect(client.hederaClient).toBeDefined();
});
});
describe('resource management', () => {
test('should close client connection', async () => {
process.env.OPERATOR_ID = '0.0.123456';
process.env.OPERATOR_KEY = '302e020100300506032b657004220420...';
await client.initialize();
await client.close();
// No errors should be thrown
expect(client.hederaClient).toBeDefined();
});
});
});

Integration Testing on Testnet

Integration tests verify that your DApp components work together correctly on a live network. We'll test on testnet to ensure real-world functionality.

Integration Test Setup

# tests/integration/dapp-flow.test.js
const HederaClient = require('../../src/client');
const TransactionHandler = require('../../src/transactions');
const { AccountId } = require('@hashgraph/sdk');
// Integration test with real Hedera testnet
describe('DApp Integration Flow', () => {
let client;
let transactionHandler;
let topicId;
beforeAll(async () => {
// Initialize test client
client = new HederaClient();
await client.initialize();
transactionHandler = new TransactionHandler(client.getClient());
});
afterAll(async () => {
await client.close();
});
test('should complete full DApp workflow', async () => {
// Step 1: Check initial balance
const initialBalance = await transactionHandler.checkBalance(client.operatorId);
expect(initialBalance).toBeDefined();
expect(initialBalance.hbars).toBeDefined();
// Step 2: Create HCS topic
topicId = await transactionHandler.createTopic('Integration Test Topic');
expect(topicId).toBeDefined();
expect(topicId.toString()).toMatch(/^0\.0\.\d+$/);
// Step 3: Send multiple messages
const messages = [
'Integration test message 1',
'Integration test message 2'
];
for (const message of messages) {
const receipt = await transactionHandler.sendMessage(topicId, message);
expect(receipt.status.toString()).toBe('SUCCESS');
}
// Step 4: Verify final balance
const finalBalance = await transactionHandler.checkBalance(client.operatorId);
expect(finalBalance).toBeDefined();
// Verify balance decreased (transaction fees paid)
const initialHbars = parseFloat(initialBalance.hbars.toString());
const finalHbars = parseFloat(finalBalance.hbars.toString());
expect(finalHbars).toBeLessThan(initialHbars);
});
test('should handle errors gracefully', async () => {
// Test with invalid topic ID
const invalidTopicId = AccountId.fromString('0.0.999999');
await expect(
transactionHandler.sendMessage(invalidTopicId, 'Test message')
).rejects.toThrow();
});
});
Integration Testing Tips
  • • Use separate test accounts for isolation
  • • Test on testnet, not mainnet
  • • Verify transactions on HashScan
  • • Test error conditions and edge cases
  • • Clean up resources after tests

Monitoring Setup

# monitoring/health-check.js
const axios = require('axios');
const { Client, AccountId } = require('@hashgraph/sdk');
class HealthMonitor {
constructor(client, config) {
this.client = client;
this.config = config;
this.isHealthy = true;
}
async checkHealth() {
try {
// Check account balance
const balance = await new AccountBalanceQuery()
.setAccountId(this.config.operatorId)
.execute(this.client);
// Check if balance is above threshold
const balanceHbar = balance.hbars.to('hbars');
if (balanceHbar < this.config.minBalance) {
throw new Error('Low balance warning');
}
// Send health check webhook
if (this.config.webhookUrl) {
await axios.post(this.config.webhookUrl, {
status: 'healthy',
balance: balanceHbar.toString(),
timestamp: new Date().toISOString()
});
}
this.isHealthy = true;
return { healthy: true, balance: balanceHbar.toString() };
} catch (error) {
console.error('Health check failed:', error.message);
this.isHealthy = false;
return { healthy: false, error: error.message };
}
}
}
module.exports = HealthMonitor;
