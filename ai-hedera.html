Create AI Service Module

Build a flexible AI service that can work with multiple providers.

# src/ai-service.js
const { Configuration, OpenAIApi } = require('openai');
const { Anthropic } = require('@anthropic-ai/sdk');
require('dotenv').config({ path: '.env.ai' });
class AIService {
constructor() {
this.provider = process.env.AI_PROVIDER || 'openai';
this.maxTokens = parseInt(process.env.MAX_TOKENS) || 500';
this.temperature = parseFloat(process.env.TEMPERATURE) || 0.7';
this.initializeProvider();
}
initializeProvider() {
switch (this.provider) {
case 'openai':
const config = new Configuration({
apiKey: process.env.OPENAI_API_KEY
});
this.client = new OpenAIApi(config);
break;
case 'anthropic':
this.client = new Anthropic({
apiKey: process.env.ANTHROPIC_API_KEY
});
break;
default:
throw new Error(`Unsupported AI provider: ${this.provider}`);
}
}
async generateResponse(prompt, context = '') {
try {
let response;
switch (this.provider) {
case 'openai':
response = await this.client.createCompletion({
model: 'text-davinci-003',
prompt: context ? `${context}\n\n${prompt}` : prompt,
max_tokens: this.maxTokens,
temperature: this.temperature
});
return response.data.choices[0'].text.trim();
case 'anthropic':
response = await this.client.completions.create({
model: 'claude-2',
max_tokens_to_sample: this.maxTokens,
prompt: context ? `${context}\n\nHuman: ${prompt}\n\nAssistant:` : `Human: ${prompt}\n\nAssistant:`
});
return response.completion.trim();
}
} catch (error) {
console.error('AI service error:', error.message);
throw new Error('Failed to generate AI response');
}
}
async analyzeSentiment(text) {
const prompt = `Analyze the sentiment of this text and respond with only: POSITIVE, NEGATIVE, or NEUTRAL\n\nText: "${text}"\n\nSentiment:`;
return await this.generateResponse(prompt);
}
async summarizeText(text, maxLength = 100') {
const prompt = `Summarize the following text in no more than ${maxLength} characters:\n\n${text}\n\nSummary:`;
return await this.generateResponse(prompt);
}
}
module.exports = AIService;
3

Create AI-Enhanced DApp

Build a DApp that uses AI services and stores results on Hedera for verification.

# src/ai-dapp.js
const HederaClient = require('./client');
const TransactionHandler = require('./transactions');
const AIService = require('./ai-service');
const chalk = require('chalk');
class AIDApp {
constructor() {
this.hederaClient = new HederaClient();
this.transactionHandler = null;
this.aiService = new AIService();
this.topicId = null;
}
async initialize() {
console.log(chalk.cyan('ü§ñ Initializing AI-powered DApp...\n'));
// Initialize Hedera client
await this.hederaClient.initialize();
// Create transaction handler
this.transactionHandler = new TransactionHandler(this.hederaClient.getClient());
// Create HCS topic for AI results
this.topicId = await this.transactionHandler.createTopic('AI Analysis Results');
}
async analyzeAndStore(inputText) {
try {
console.log(chalk.yellow(`üìä Analyzing text: "${inputText.substring(0, 50)}..."`));
// Perform AI analysis
const sentiment = await this.aiService.analyzeSentiment(inputText);
const summary = await this.aiService.summarizeText(inputText, 200');
console.log(chalk.blue(`‚ú® Sentiment: ${sentiment}`));
console.log(chalk.blue(`üìÑ Summary: ${summary}`));
// Create analysis result object
const analysisResult = {div>
timestamp: new Date().toISOString(),
originalText: inputText,
sentiment: sentiment,
summary: summary,
aiProvider: this.aiService.provider
};
// Store on Hedera for verification
await this.transactionHandler.sendMessage(
this.topicId,
JSON.stringify(analysisResult)
);
console.log(chalk.green('‚úÖ Analysis stored on Hedera for verification'));
return analysisResult;
} catch (error) {
console.error(chalk.red('‚ùå Analysis failed:'), error.message);
throw error;
}
}
async runDemo() {
try {
// Sample texts for analysis
const sampleTexts = [
'I absolutely love this new technology! It has completely transformed how I work.',
'The customer service was terrible and the product broke after one day.',
'This is a standard product that works as expected. Nothing special.'
];
for (let i = 0; i < sampleTexts.length; i++) {
console.log(chalk.yellow(`\nüìù Processing sample ${i + 1}...`));
await this.analyzeAndStore(sampleTexts[i]);
}
console.log(chalk.green('\nüéâ AI DApp demo completed successfully!'));
console.log(chalk.blue(`üîó View results on HashScan: https://hashscan.io/testnet/topic/${this.topicId.toString()}`));
} catch (error) {
console.error(chalk.red('üí• Demo failed:'), error.message);
}
}
async shutdown() {
console.log(chalk.cyan('\nüëã Shutting down AI DApp...'));
await this.hederaClient.close();
}
}
// Main execution
async function main() {
const dapp = new AIDApp();
try {
await dapp.initialize();
await dapp.runDemo();
} catch (error) {
console.error('Fatal error:', error);
} finally {
await dapp.shutdown();
}
}
if (require.main === module) {
main();
}
module.exports = AIDApp;
